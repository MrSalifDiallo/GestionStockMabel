image: php:8.2

stages:
  - build
  - test
  - deploy

cache:
  paths:
    - vendor/

before_script:
  - apt-get update -qq && apt-get install -y -qq git zip unzip
  - curl -sS https://getcomposer.org/installer | php -- --install-dir=/usr/local/bin --filename=composer

# ========== STAGE 1: BUILD ==========
build:dependencies:
  stage: build
  script:
    - composer install --no-progress --prefer-dist
  artifacts:
    paths:
      - vendor/
    expire_in: 1 week

# ========== STAGE 2: TEST ==========
test:phpunit:
  stage: test
  needs:
    - build:dependencies
  services:
    - mysql:5.7
  variables:
    MYSQL_ROOT_PASSWORD: "root"
    MYSQL_DATABASE: "laravel_test"
    MYSQL_USER: "laravel"
    MYSQL_PASSWORD: "laravel"
    DB_HOST: "mysql"
    DB_DATABASE: "laravel_test"
    DB_USERNAME: "laravel"
    DB_PASSWORD: "laravel"
  before_script:
    - apt-get update -qq && apt-get install -y -qq git zip unzip mysql-client
    - curl -sS https://getcomposer.org/installer | php -- --install-dir=/usr/local/bin --filename=composer
    - cp .env.example .env.testing
  script:
    - composer install --no-progress --prefer-dist
    - php artisan key:generate --env=testing
    - php artisan migrate --env=testing --force
    - vendor/bin/phpunit
  allow_failure: false

# ========== STAGE 3: DEPLOY VIA FTP ==========
deploy:lws_ftp:
  stage: deploy
  needs:
    - build:dependencies
    - test:phpunit
  only:
    - main
  before_script:
    - apt-get update -qq && apt-get install -y -qq lftp
  script:
    - |
      lftp -c "
      set ftp:ssl-allow no;
      open -u $FTP_USERNAME,$FTP_PASSWORD $FTP_HOST;
      mirror -Rev . $FTP_TARGET_DIR \
        --ignore-time \
        --parallel=5 \
        --exclude-glob .git* \
        --exclude .git/ \
        --exclude .env \
        --exclude .env.example \
        --exclude .env.testing \
        --exclude vendor/ \
        --exclude node_modules/ \
        --exclude tests/ \
        --exclude storage/logs/* \
        --exclude storage/framework/sessions/* \
        --exclude storage/framework/cache/*;
      quit
      "
  after_script:
    - echo "Files uploaded to LWS via FTP"
